# อธิบายการทำงาน Frontend แบบละเอียด (พร้อมโค้ด)

> **Stack:** Next.js 15 + React 19 + TypeScript + CSS Modules + Tailwind CSS  
> **Port:** 3000  
> **หน้าที่หลัก:** รับ input จาก user → เรียก NestJS Backend → แสดงผลวิเคราะห์ตลาด

---

## โครงสร้างไฟล์

```
frontend/
└── src/
    └── app/
        ├── layout.tsx        ← Root layout (HTML wrapper, fonts)
        ├── page.tsx          ← หน้าหลัก (ทั้ง app อยู่ที่นี่)
        ├── globals.css       ← Global styles + Tailwind import
        └── page.module.css   ← CSS Modules สำหรับ page.tsx
```

// page.tsx บรรทัด 70-81

export default function HomePage() {
  // ── Form inputs ──────────────────────────────────────────────────
  const [topic, setTopic]     = useState('');           // text input ของ topic
  const [region, setRegion]   = useState('');           // region ที่เลือกใน dropdown

### `handleRegionChange()` — เปลี่ยน region

```typescript
// page.tsx บรรทัด 83-87

const handleRegionChange = (value: string) => {
  setRegion(value);     // อัปเดต region
  setMarkets([]);       // ← clear markets เพราะ market ของ region เก่าไม่ valid แล้ว
  setError('');         // clear error
};
```

> ทุกครั้งที่เปลี่ยน region → markets reset เป็น `[]` ทันที  
> `availableMarkets` จะ recalculate อัตโนมัติตาม region ใหม่

### `handleMarketToggle()` — toggle เลือก/ยกเลิก market

```typescript
// page.tsx บรรทัด 89-94

const handleMarketToggle = (market: string) => {
  setError('');
  setMarkets((prev) =>
    prev.includes(market)
      ? prev.filter((item) => item !== market)  // ถ้ามีอยู่แล้ว → ลบออก (deselect)
      : [...prev, market]                        // ถ้ายังไม่มี → เพิ่มเข้า (select)
  );
};
```

> ใช้ functional update `(prev) => ...` เพราะ state update อาจ batch ได้  
> เป็น multi-select — กดซ้ำ = toggle off

### `loadHistory()` — โหลดประวัติจาก backend

```typescript
// page.tsx บรรทัด 96-115

const loadHistory = async () => {
  try {
    setHistoryLoading(true);  // แสดง "Loading..." ใน sidebar

    // เรียก GET /market/history
    const response = await fetch(`${apiUrl}/market/history`, {
      cache: 'no-store',  // ← ห้าม browser cache ผล เพื่อให้ได้ข้อมูลล่าสุดเสมอ
    });

    if (!response.ok) {
      throw new Error('Failed to load history.');
    }

    const data: HistoryItem[] = await response.json();
    setHistory(Array.isArray(data) ? data : []);  // ป้องกันกรณี response ไม่ใช่ array
  } catch (err) {
    console.error('History load error:', err);  // ไม่แสดง error ให้ user เห็น (silent fail)
  } finally {
    setHistoryLoading(false);  // ปิด loading ไม่ว่าสำเร็จหรือไม่
  }
};

// เรียก loadHistory() ครั้งแรกตอน component mount
useEffect(() => {
  loadHistory();
}, []);  // [] = รันแค่ครั้งเดียว
```

### `handleAnalyze()` — ส่งข้อมูลวิเคราะห์ (function หลัก)

```typescript
// page.tsx บรรทัด 121-165

const handleAnalyze = async () => {
  const trimmedTopic = topic.trim();

  // ── Client-side validation ────────────────────────────────────────
  if (!trimmedTopic) {
    setError('Please enter a market topic.');
    return;  // หยุดทันที ไม่เรียก API
  }
  if (!region) {
    setError('Please select a region.');
    return;
  }
  if (markets.length === 0) {
    setError('Please choose at least one market.');
    return;
  }

  setLoading(true);
  setError('');

  try {
    // ── เรียก POST /market/analyze ───────────────────────────────────
    const response = await fetch(`${apiUrl}/market/analyze`, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({
        topic: trimmedTopic,
        region,
        markets,     // string[]
      }),
    });

    if (!response.ok) {
      throw new Error('Failed to analyze market.');
    }

    const data: AnalyzeResponse = await response.json();
    setResult(data);          // ← แสดงผลในหน้า content
    await loadHistory();      // ← refresh ประวัติใน sidebar ทันที

  } catch (err) {
    // แสดง error ให้ user เห็น
    setError(err instanceof Error ? err.message : 'Something went wrong.');
  } finally {
    setLoading(false);  // ปิด loading button
  }
};
```

### `handleSelectHistory()` — กดเลือกประวัติใน sidebar

```typescript
// page.tsx บรรทัด 167-211

const handleSelectHistory = (item: HistoryItem) => {
  try {
    // ── แปลง HistoryItem → AnalyzeResponse ───────────────────────────
    // เพราะ markets ใน DB อาจเป็น string หรือ array (SQLite simple-json quirk)
    const mappedResult: AnalyzeResponse = {
      topic: item.topic,
      region: item.region,
      markets: Array.isArray(item.markets)
        ? item.markets  // ถ้าเป็น array → ใช้เลย
        : typeof item.markets === 'string'
          ? item.markets.split(',').map((v) => v.trim()).filter(Boolean)  // CSV → array
          : [],
      keyMarkets:         item.keyMarkets ?? [],
      marketInsights:     item.marketInsights ?? [],
      recentDevelopments: item.recentDevelopments ?? [],
      externalSignals:    item.externalSignals ?? [],
      overallInsight:     item.overallInsight ?? '',
      opportunities:      item.opportunities ?? [],
      risks:              item.risks ?? [],
    };

    setResult(mappedResult);    // ← แสดงผลเก่าใน content area

    // ── Sync form fields กับ history item ────────────────────────────
    setTopic(item.topic);
    setRegion(item.region);

    // parse markets เหมือนกัน
    if (Array.isArray(item.markets)) {
      setMarkets(item.markets);
    } else if (typeof item.markets === 'string') {
      setMarkets(item.markets.split(',').map((v) => v.trim()).filter(Boolean));
    } else {
      setMarkets([]);
    }

    setError('');
  } catch (err) {
    console.error('Failed to load history result:', err);
    setError('Unable to load this history item.');
  }
};
```

// page.tsx บรรทัด 213-226

// คำนวณ markets ที่จะแสดงใน result card

## Flow รวม User Interaction

```
User เปิดหน้า
        │
        ▼
[useEffect → loadHistory()]
  GET /market/history
  → setHistory(data)
  → แสดงประวัติใน sidebar
        │
User พิมพ์ topic / กด chip
        │
        ▼
[onChange → setTopic(value)]
        │
User เลือก region
        │
        ▼
[handleRegionChange(value)]
  → setRegion(value)
  → setMarkets([])       ← reset markets
  → availableMarkets recalculates
        │
User กด market chips (multi-select)
        │
        ▼
[handleMarketToggle(market)]
  → setMarkets(toggle)
        │
User กด "Analyze Market"
        │
        ▼
[handleAnalyze()]
  ├── Validate topic, region, markets
  │   ถ้าไม่ผ่าน → setError() → หยุด
  │
  ├── setLoading(true)
  │
  ├── POST /market/analyze { topic, region, markets }
  │         │
  │         ▼
  │   NestJS Backend → Python AI → กลับมา
  │         │
  │         ▼
  ├── setResult(data)     ← แสดง 7 result cards
  ├── loadHistory()       ← refresh sidebar
  └── setLoading(false)
        │
User กด history item
        │
        ▼
[handleSelectHistory(item)]
  → แปลง HistoryItem → AnalyzeResponse
  → setResult(mappedResult)   ← แสดงผลเก่า
  → setTopic / setRegion / setMarkets  ← sync form
```
